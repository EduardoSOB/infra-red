#include <IRremote.h>
int receiver = 13;
IRrecv irrecv(receiver);
decode_results results;

void setup()
{
  Serial.begin(115200);
  irrecv.enableIRIn();
}
void loop()
{
  if(irrecv.decode(&results)){
    Serial.println (results.value, HEX);
    irrecv.resume();
    }
     if(results.value == 0x4040708F){
      digitalWrite(7,HIGH);
      }
}
